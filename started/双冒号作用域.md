# 示例代码

局部：

```c++
#include <vector>
#include <iostream>

int a = 10;

int main() {
  int a = 20;
  std::cout << a << std::endl; // 输出局部的 20
  return 0;
}
```

全局：

```c++
#include <vector>
#include <iostream>

int a = 10;

int main() {
  int a = 20;
  std::cout << ::a << std::endl; // 输出局部的 10
  return 0;
}
```

namespace

```c++
#include <vector>
#include <iostream>

int a = 10;

namespace A {
  int a = 30;
}

int main() {
  int a = 20;
  std::cout << A::a << std::endl; // 输出局部的 30
  return 0;
}
```

class

```c++

#include <vector>
#include <iostream>

int a = 10;

namespace A {
  int a = 30;
}

class D {
  public:
    static int a;
};

int D::a = 100;

int main() {
  int a = 20;
  D d1;
  std::cout << d1.a << std::endl; // 输出局部的 100
  return 0;
}
```
